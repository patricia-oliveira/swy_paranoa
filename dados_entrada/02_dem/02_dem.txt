// Import the dataset and select the elevation band.
var dataset = ee.Image('NASA/NASADEM_HGT/001');
var elevation = dataset.select('elevation').clip(roi);

// Add a white background image to the map.
var background = ee.Image(1);

Map.addLayer(background, {min: 0, max: 1});

// Set elevation visualization properties.
var elevationVis = {
  min: 800,
  max: 1200,
};

// Set elevation <= 0 as transparent and add to the map.
Map.addLayer(elevation, elevationVis, 'Elevation'); //elevation.updateMask(elevation.gt(0))
Map.centerObject(roi);

// Exportação para o Google Drive
Export.image.toDrive({
  image: elevation, 
  description: 'dem',          
  folder: 'Pati_Exports',                    
  fileNamePrefix: 'dem' ,        
  region: roi,                                  
  scale: 30,                                     
  crs: 'EPSG:4326',                             
  fileFormat: 'GeoTIFF',                         
  
});
